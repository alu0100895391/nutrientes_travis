 def aibc i
   s = []
       datos[i][1..datos[i].length-1].zip(datos[i][0..datos[i].length-2]) do    |x,y|
               if x < datos[i][0]
                       s << 0.0
               else
                       s << (((x-datos[i][0])+(y-datos[i][0]))/2)*5 

               end
       end
               
       s.reduce(:+)
